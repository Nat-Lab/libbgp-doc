193 ssize_t BgpPathAttrib::parseHeader(const uint8_t *from, size_t buffer_sz) {
194  if (buffer_sz < 3) {
195  setError(E_UPDATE, E_UNSPEC_UPDATE, NULL, 0);
196  logger->log(ERROR, "BgpPathAttrib::parseHeader: invalid attribute header size.\n");
197  return -1;
198  }
199  const uint8_t *buffer = from;
200 
201  uint8_t flags = getValue<uint8_t>(&buffer);
202 
203  optional = (flags >> 7) & 0x1;
204  transitive = (flags >> 6) & 0x1;
205  partial = (flags >> 5) & 0x1;
206  extended = (flags >> 4) & 0x1;
207  type_code = getValue<uint8_t>(&buffer);
208  if (extended && buffer_sz < 4) {
209  setError(E_UPDATE, E_UNSPEC_UPDATE, NULL, 0);
210  logger->log(ERROR, "BgpPathAttrib::parseHeader: invalid attribute header size (extended but size < 4).\n");
211  return -1;
212  }
213 
214  if (extended) value_len = ntohs(getValue<uint16_t>(&buffer));
215  else value_len = getValue<uint8_t>(&buffer);
216 
217  if (value_len > buffer_sz - 3) {
218  err_code = E_UPDATE;
219  // This is kind of "invalid length", but we are not using E_ATTR_LEN.
220  // E_ATTR_LEN: "Attribute Length that conflict with the expected length
221  // (based on the attribute type code)." This is not based on type code,
222  // but it is buffer overflow, so we set subcode to E_UNSPEC,
223  setError(E_UPDATE, E_UNSPEC_UPDATE, NULL, 0);
224  logger->log(ERROR, "BgpPathAttrib::parseHeader: value_length (%d) < buffer left (%d).\n", value_len, buffer_sz - 3);
225  return -1;
226  }
227 
228  return extended ? 4 : 3;
229 }
