115 const BgpRib6Entry* BgpRib6::insert(BgpLogHandler *logger, const Prefix6 &route,
116  const uint8_t nexthop_global[16], const uint8_t nexthop_linklocal[16], int32_t weight) {
117  std::vector<std::shared_ptr<BgpPathAttrib>> attribs;
118  BgpPathAttribOrigin *origin = new BgpPathAttribOrigin(logger);
119  BgpPathAttribAsPath *as_path = new BgpPathAttribAsPath(logger, true);
120  origin->origin = IGP;
121 
122  attribs.push_back(std::shared_ptr<BgpPathAttrib>(origin));
123  attribs.push_back(std::shared_ptr<BgpPathAttrib>(as_path));
124 
125  BgpRib6Entry new_entry(route, 0, nexthop_global, nexthop_linklocal, attribs);
126  new_entry.weight = weight;
127  uint64_t use_update_id = update_id;
128 
129  for (const auto &entry : rib) {
130  if (entry.second.src_router_id == 0 && entry.second.route == route) {
131  this->logger->log(ERROR, "BgpRib6::insert: route exists.\n");
132  return NULL;
133  }
134 
135  // see if we can group this entry to other local entries
136  if (entry.second.src_router_id == 0) {
137  if (memcmp(new_entry.nexthop_global, entry.second.nexthop_global, 16) == 0 &&
138  memcmp(new_entry.nexthop_linklocal, entry.second.nexthop_linklocal, 16) == 0) {
139  use_update_id = entry.second.update_id;
140  }
141  }
142  }
143 
144  std::lock_guard<std::recursive_mutex> lock(mutex);
145  new_entry.update_id = use_update_id;
146  if (use_update_id == update_id) update_id++;
147  rib6_t::const_iterator it = rib.insert(MAKE_ENTRY6(route, 0, new_entry));
148 
149  return &(it->second);
150 }
