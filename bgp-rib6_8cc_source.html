118 
119  BgpRib6Entry new_entry(route, 0, nexthop_global, nexthop_linklocal, attribs);
120  uint64_t use_update_id = update_id;
121 
122  for (const BgpRib6Entry &entry : rib) {

126  }

138  new_entry.update_id = use_update_id;
