273 const BgpRib6Entry* BgpRib6::selectEntry(const BgpRib6Entry *a, const BgpRib6Entry *b) {
274  if (a == NULL) return b;
275  if (b == NULL) return a;
276 
277  const Route6 &ra = a->route;
278  const Route6 &rb = b->route;
279 
280  // a is more specific, use a
