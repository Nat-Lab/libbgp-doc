175 int BgpFsm::run(const uint8_t *buffer, const size_t buffer_size) {
176  if (state == BROKEN) {
177  logger->log(ERROR, "BgpFsm::run: FSM is broken, consider reset.\n");
178  return -1;
179  }
180 
181  ssize_t fill_ret = in_sink.fill(buffer, buffer_size);
182  if (fill_ret != (ssize_t) buffer_size) {
183  logger->log(ERROR, "BgpFsm::run: failed to fill() sink.\n");
184  setState(BROKEN);
185  return -1;
186  }
187 
188  // tick the clock
189  int tick_ret = tick();
190  if (tick_ret <= 0) return tick_ret;
191  last_recv = clock->getTime();
192 
193  int final_ret_val = -1;
194 
195  // keep running untill sink empty
196  while (in_sink.getBytesInSink() > 0) {
197  BgpPacket *packet = NULL;
198  ssize_t poured = in_sink.pour(&packet);
199 
200  if (poured <= -2) {
201  logger->log(ERROR, "BgpFsm::run: sink seems to be broken, please reset.\n");
202  setState(BROKEN);
203  return -1;
204  }
205 
206  if (poured == 0) return 3;
207 
208  LIBBGP_LOG(logger, DEBUG) {
209  logger->log(DEBUG, "BgpFsm::run: got message (Current state: %s):\n", bgp_fsm_state_str[state]);
210  logger->log(DEBUG, *packet);
211  }
212 
213  const BgpMessage *msg = packet->getMessage();
214  // parse failed / packet invalid (errors like Unsupported Optional
215  // Parameter falls in this catagory, since those errors are checked by
216  // parsers, other errors like FSM error, Bad Peer AS, etc is handled in
217  // fsmEval*)
218  if (poured == -1) {
219  if (msg->type == NOTIFICATION) {
220  logger->log(ERROR, "BgpFsm::run: got invalid NOTIFICATION message.\n");
221  if (state == ESTABLISHED) {
222  logger->log(ERROR, "BgpFsm::run: discarding all routes.\n");
223  }
224 
225  delete packet;
226  setState(IDLE);
227  return 0;
228  }
229  BgpNotificationMessage notify (logger, msg->getErrorCode(), msg->getErrorSubCode(), msg->getError(), msg->getErrorLength());
230  setState(IDLE);
231  if(!writeMessage(notify)) return -1;
232  delete packet;
233  return 0;
234  }
235 
236  if (msg->type == NOTIFICATION) {
237  const BgpNotificationMessage *notify = dynamic_cast<const BgpNotificationMessage *>(msg);
238  const char *err_msg = bgp_error_code_str[notify->errcode];
239  const char *err_sub_msg = bgp_error_code_str[0];
240  switch (notify->errcode) {
241  case E_HEADER: err_sub_msg = bgp_header_error_subcode_str[notify->subcode]; break;
242  case E_OPEN: err_sub_msg = bgp_open_error_subcode_str[notify->subcode]; break;
243  case E_UPDATE: err_sub_msg = bgp_update_error_str[notify->subcode]; break;
244  case E_FSM: err_sub_msg = bgp_fsm_error_str[notify->subcode]; break;
245  case E_CEASE: err_sub_msg = bgp_cease_error_str[notify->subcode]; break;
246  }
247  logger->log(ERROR, "BgpFsm::run: got NOTIFICATION: %s (%d): %s (%d).\n", err_msg, notify->errcode, err_sub_msg, notify->subcode);
248  delete packet;
249  setState(IDLE);
250  return 0;
251  }
252 
253  int retval = -1;
254 
255  int vald_ret = validateState(msg->type);
256  if (vald_ret <= 0) {
257  delete packet;
258  return vald_ret;
259  }
260 
261  switch (state) {
262  case IDLE: retval = fsmEvalIdle(msg); break;
263  case OPEN_SENT: retval = fsmEvalOpenSent(msg); break;
264  case OPEN_CONFIRM: retval = fsmEvalOpenConfirm(msg); break;
265  case ESTABLISHED: retval = fsmEvalEstablished(msg); break;
266  default: {
267  logger->log(ERROR, "BgpFsm::run: FSM in invalid state: %d.\n", state);
268  delete packet;
269  return -1;
270  }
271  }
272 
273  delete packet;
274  if (retval < 0) return retval;
275  if (retval == 0) final_ret_val = 0;
276  if (retval == 1 && final_ret_val != 0 && final_ret_val != 2) final_ret_val = 1;
277  if (retval == 2) final_ret_val = 2;
278  }
279 
280  return final_ret_val;
281 }

317 int BgpFsm::openRecv(const BgpOpenMessage *open_msg) {
318  if (open_msg->version != 4) {
319  BgpNotificationMessage notify (logger, E_OPEN, E_VERSION, NULL, 0);
320  setState(IDLE);
321  if(!writeMessage(notify)) return -1;
322  return 0;
323  }
324 
325  if (config.peer_asn != 0 && open_msg->my_asn != config.peer_asn) {
326  BgpNotificationMessage notify (logger, E_OPEN, E_PEER_AS, NULL, 0);
327  setState(IDLE);
328  if(!writeMessage(notify)) return -1;
329  return 0;
330  }
331 
332  if (!validAddr(open_msg->bgp_id)) {
333  LIBBGP_LOG(logger, ERROR) {
334  char ip_str[INET_ADDRSTRLEN];
335  inet_ntop(AF_INET, &(open_msg->bgp_id), ip_str, INET_ADDRSTRLEN);
336  logger->log(ERROR, "BgpFsm::openRecv: peer BGP ID (%s) invalid.\n", ip_str);
337  }
338  BgpNotificationMessage notify (logger, E_OPEN, E_BGP_ID, NULL, 0);
339  setState(IDLE);
340  if(!writeMessage(notify)) return -1;
341  return 0;
342  }
343 
344  // if hold timer != 0 but < 3, reject witl E_HOLD_TIME (as per rfc4271)
345  if (open_msg->hold_time < 3 && open_msg->hold_time != 0) {
346  logger->log(ERROR, "BgpFsm::openRecv: invalid hold timer %d.\n", open_msg->hold_time);
347  BgpNotificationMessage notify (logger, E_OPEN, E_HOLD_TIME, NULL, 0);
348  setState(IDLE);
349  if(!writeMessage(notify)) return -1;
350  return 0;
351  }
352 
353  if (!config.no_collision_detection && rev_bus_exist) {
354  RouteCollisionEvent col = RouteCollisionEvent();
355  col.peer_bgp_id = open_msg->bgp_id;
356 
357  // publish returned 0: no one complained about collision (either
358  // there's no collision, or some fsm has killed themselves)
359  //
360  // publish returned > 0: someone complained about collision, it think
361  // this session should be dropped
362  if(config.rev_bus->publish(this, col) > 0) {
363  int res_result = resloveCollision(open_msg->bgp_id, true);
364 
365  if(res_result == -1) return -1;
366  if(res_result == 0) return 0;
367 
368  // should not be happening
369  if(res_result == 1) {
370  logger->log(ERROR,
371  "BgpFsm::openRecv: collision found, and some other FSM feels like they should live"
372  "while we feel like we should live too. is there duplicated FSMs?\n"
373  );
374  setState(BROKEN);
375  return -1;
376  }
377  }
378  }
379 
380  hold_timer = config.hold_timer > open_msg->hold_time ? open_msg->hold_time : config.hold_timer;
381  peer_bgp_id = open_msg->bgp_id;
382  peer_asn = open_msg->getAsn();
383  use_4b_asn = open_msg->hasCapability(ASN_4B) && config.use_4b_asn;
384  if (open_msg->hasCapability(MP_BGP)) {
385  send_ipv4_routes = send_ipv6_routes = false;
386 
387  const std::vector<std::shared_ptr<BgpCapability>> &capabilities = open_msg->getCapabilities();
388 
389  for (const std::shared_ptr<BgpCapability> &cap : capabilities) {
390  if (cap->code == MP_BGP) {
391  const BgpCapabilityMpBgp &mp_cap = dynamic_cast<const BgpCapabilityMpBgp &> (*cap);
392  if (mp_cap.safi != UNICAST) continue;
393  if (mp_cap.afi == IPV6) send_ipv6_routes = true && config.mp_bgp_ipv6;
394  if (mp_cap.afi == IPV4) send_ipv4_routes = true && config.mp_bgp_ipv4;
395  }
396  }
397  } else {
398  send_ipv4_routes = true && !(config.mp_bgp_ipv6 && !config.mp_bgp_ipv4);
399  send_ipv6_routes = false;
400  }
401 
402  return 1;
403 }

405 int BgpFsm::resloveCollision(uint32_t peer_bgp_id, bool is_new) {
406  if(is_new) {
407  if (ntohl(config.router_id) > ntohl(peer_bgp_id)) {
408  // this is a new connection, and "we" have higer ID, there's
409  // already a connection so THIS fsm should be dispose. since
410  // THIS fsm is created by peer connecting to us.
411  BgpNotificationMessage notify (logger, E_CEASE, E_COLLISION, NULL, 0);
412  if(!writeMessage(notify)) return -1;
413 
414  logger->log(INFO, "(new_session) we have higher router id, dispose this new session.\n");
415 
416  setState(IDLE);
417  return 0;
418  } else {
419  // this is a new connection, and peer has higher ID. the exisiting
420  // connection should be dispose, since THIS fsm is created by peer
421  // connecting to us, this one will be kept, we do nothing.
422 
423  logger->log(INFO, "(new_session) peer has higher router id, ask the other fsm to dispose session.\n");
424  return 1;
425  }
426  } else {
427  if (ntohl(config.router_id) > ntohl(peer_bgp_id)) {
428  // this is a old connection, and "we" have higer ID, the new one
429  // shoud close, we do nothing.
430  logger->log(INFO, "(old_session) we have higher router id, keep this old session.\n");
431 
432  return 1;
433  } else {
434  // this is a old connection, and "peer" have higer ID, this one
435  // shoud close.
436 
437  logger->log(INFO, "(old_session) peer has higher router id, dispose this session.\n");
438  BgpNotificationMessage notify (logger, E_CEASE, E_COLLISION, NULL, 0);
439  if(!writeMessage(notify)) return -1;
440 
441  setState(IDLE);
442  return 0;
443  }
444  }
445 
446  // UNREACHED
447  setState(BROKEN);
448  logger->log(ERROR, "BgpFsm::resloveCollison: ??? :( \n");
449  return -1;
450 }

540  }
541  }

560  return 0;
561  }
562  return 1;
563  case OPEN_SENT:
564  if (type != OPEN) {
565  logger->log(ERROR, "BgpFsm::validateState: got non-OPEN message in OPEN_SENT state.\n");

567  setState(IDLE);

569

571  }
572  return 1;
573  case OPEN_CONFIRM:
574  if (type != KEEPALIVE) {

576  BgpNotificationMessage notify (logger, E_FSM, E_OPEN_CONFIRM, NULL, 0);
577  setState(IDLE);

579 
580  return 0;
581  }

583  case ESTABLISHED:

585  logger->log(ERROR, "BgpFsm::validateState: got invalid message (type %d) in ESTABLISHED state.\n", type);
586  BgpNotificationMessage notify (logger, E_FSM, E_ESTABLISHED, NULL, 0);
587  setState(IDLE);
588  if(!writeMessage(notify)) return -1;
589 
590  return 0;
591  }
592  return 1;
593  default:

595  return -1;
596  }

645 int BgpFsm::fsmEvalOpenConfirm(__attribute__((unused)) const BgpMessage *msg) {
646  BgpKeepaliveMessage keep = BgpKeepaliveMessage(logger);
647  setState(ESTABLISHED);
648  if(!writeMessage(keep)) return -1;
649 
650  if (send_ipv4_routes) {
651  std::vector<BgpRib4Entry>::const_iterator iter = rib4->get().begin();
652  const std::vector<BgpRib4Entry>::const_iterator end = rib4->get().end();
653 
654  // group routes and and updates
655  while (iter != end) {
656  uint64_t cur_group_id = iter->update_id;
657  BgpUpdateMessage update (logger, use_4b_asn);
658  update.setAttribs(iter->attribs);
659  prepareUpdateMessage(update, true);
660 
661  // length of the update message, 19: headers, 4: length fields
662  size_t msg_len = 19 + 4;
663 
664  for (const std::shared_ptr<BgpPathAttrib> &attrib : update.path_attribute) {
665  msg_len += attrib->length();
666  }
667 
668  for (; iter != end && cur_group_id == iter->update_id && msg_len < 4096; iter++) {
669  const Prefix4 &r = iter->route;
670  if (config.out_filters4.apply(r.getPrefix(), r.getLength()) == ACCEPT) {
671  msg_len += 1 + (r.getLength() + 7) / 8;
672  if (msg_len > 4096) {
673  // size too big, roll back and break.
674  iter--;
675  break;
676  }
677  update.addNlri4(r);
678  } else {
679  LIBBGP_LOG(logger, INFO) {
680  uint32_t prefix = r.getPrefix();
681  char ip_str[INET_ADDRSTRLEN];
682  inet_ntop(AF_INET, &prefix, ip_str, INET_ADDRSTRLEN);
683  logger->log(INFO, "BgpFsm::fsmEvalOpenConfirm: route %s/%d filtered by out_filter.\n", ip_str, r.getLength());
684  }
685  }
686  }
687 
688  if (update.nlri.size() > 0) {
689  if(!writeMessage(update)) return -1;
690  }
691  }
692  }
693 
694  if (send_ipv6_routes) {
695  std::vector<BgpRib6Entry>::const_iterator iter = rib6->get().begin();
696  std::vector<BgpRib6Entry>::const_iterator end = rib6->get().end();
697 
698  while (iter != end) {
699  uint64_t cur_group_id = iter->update_id;
700  const uint8_t *nh_global = iter->nexthop_global;
701  const uint8_t *nh_linklocal = iter->nexthop_linklocal;
702  BgpUpdateMessage update (logger, use_4b_asn);
703  update.setAttribs(iter->attribs);
704  prepareUpdateMessage(update, false);
705  std::vector<Prefix6> filtered_nlri;
706 
707  // 8: mp-reach-nlri headers (attrib hdr: 3, afi/safi/nh_len/res: 5)
708  // 32: max nexthop len
709  size_t msg_len = 19 + 4 + 8 + 32;
710  for (const std::shared_ptr<BgpPathAttrib> &attrib : update.path_attribute) {
711  msg_len += attrib->length();
712  }
713 
714  for (; iter != end && cur_group_id == iter->update_id && msg_len < 4096; iter++) {
715  const Prefix6 &r = iter->route;
716  if (config.out_filters6.apply(r) == ACCEPT) {
717  msg_len += 1 + (r.getLength() + 7) / 8;
718  if (msg_len > 4096) {
719  // size too big, roll back and break.
720  iter--;
721  break;
722  }
723  filtered_nlri.push_back(r);
724  } else {
725  LIBBGP_LOG(logger, INFO) {
726  uint8_t prefix[16];
727  r.getPrefix(prefix);
728  char ip_str[INET6_ADDRSTRLEN];
729  inet_ntop(AF_INET6, &prefix, ip_str, INET6_ADDRSTRLEN);
730  logger->log(INFO, "BgpFsm::fsmEvalOpenConfirm: route %s/%d filtered by out_filter.\n", ip_str, r.getLength());
731  }
732  }
733  }
734 
735  if (filtered_nlri.size() > 0) {
736  if (config.forced_default_nexthop6 && !config.peering_lan6.includes(nh_global)) {
737  LIBBGP_LOG(logger, INFO) {
738  char nh_old_str[INET6_ADDRSTRLEN];
739  char nh_def_str[INET6_ADDRSTRLEN];
740  inet_ntop(AF_INET6, &nh_global, nh_old_str, INET6_ADDRSTRLEN);
741  inet_ntop(AF_INET6, &nh_def_str, nh_def_str, INET6_ADDRSTRLEN);
742 
743  if (config.forced_default_nexthop6) {
744  logger->log(INFO, "BgpFsm::fsmEvalOpenConfirm: forced_default_nexthop6 set, default (%s) will be used.\n", nh_def_str);
745  }
746  else logger->log(INFO, "BgpFsm::fsmEvalOpenConfirm: nexthop %s is not in peering lan, default (%s) will be used.\n", nh_old_str, nh_def_str);
747  }
748 
749  update.setNlri6(filtered_nlri, config.default_nexthop6_global, config.default_nexthop6_linklocal);
750  } else update.setNlri6(filtered_nlri, nh_global, nh_linklocal);
751 
752 
753  if(!writeMessage(update)) return -1;
754  }
755 
756  }
757  }
758 
759  return 1;

762 int BgpFsm::fsmEvalEstablished(const BgpMessage *msg) {
763  if (msg->type == KEEPALIVE) return 1;
764 
765  const BgpUpdateMessage *update = dynamic_cast<const BgpUpdateMessage *>(msg);
766 
767  if (send_ipv4_routes) {
768  rib4->withdraw(peer_bgp_id, update->withdrawn_routes);
769 
770  if (update->nlri.size() > 0) {
771  const BgpPathAttribNexthop &nh = dynamic_cast<const BgpPathAttribNexthop &>(update->getAttrib(NEXT_HOP));
772 
773  if (!validAddr(nh.next_hop)) {
774  LIBBGP_LOG(logger, WARN) {
775  char ip_str[INET_ADDRSTRLEN];
776  inet_ntop(AF_INET, &(nh.next_hop), ip_str, INET_ADDRSTRLEN);
777  logger->log(WARN, "BgpFsm::fsmEvalEstablished: ignored %zu routes with invalid nexthop %s\n", update->nlri.size(), ip_str);
778  }
779  return 1;
780  }
781 
782  if (!config.no_nexthop_check4 && !config.peering_lan4.includes(nh.next_hop)) {
783  LIBBGP_LOG(logger, WARN) {
784  char ip_str_nh[INET_ADDRSTRLEN];
785  char ip_str_lan[INET_ADDRSTRLEN];
786  inet_ntop(AF_INET, &(nh.next_hop), ip_str_nh, INET_ADDRSTRLEN);
787  uint32_t peering_lan_pfx = config.peering_lan4.getPrefix();
788  inet_ntop(AF_INET, &peering_lan_pfx, ip_str_lan, INET_ADDRSTRLEN);
789  logger->log(WARN, "BgpFsm::fsmEvalEstablished: ignored %zu routes with nexthop outside peering LAN. (%s not in %s/%d)\n",
790  update->nlri.size(), ip_str_nh, ip_str_lan, peering_lan_pfx);
791  }
792  return 1;
793  };
794  }
795 
796  std::vector<Prefix4> routes = std::vector<Prefix4> ();
797  for (const Prefix4 &route : update->nlri) {
798  if(config.in_filters4.apply(route.getPrefix(), route.getLength()) == ACCEPT) {
799  routes.push_back(route);
800  } else {
801  LIBBGP_LOG(logger, INFO) {
802  uint32_t prefix = route.getPrefix();
803  char ip_str[INET_ADDRSTRLEN];
804  inet_ntop(AF_INET, &prefix, ip_str, INET_ADDRSTRLEN);
805  logger->log(INFO, "BgpFsm::fsmEvalEstablished: route %s/%d filtered by in_filters4.\n", ip_str, route.getLength());
806  }
807  }
808  }
809 
810  if (routes.size() > 0) {
811  rib4->insert(peer_bgp_id, routes, update->path_attribute);
812  }
813 
814  if (rev_bus_exist) {
815  if (update->withdrawn_routes.size() > 0) {
816  Route4WithdrawEvent wev = Route4WithdrawEvent();
817  wev.routes = update->withdrawn_routes;
818  config.rev_bus->publish(this, wev);
819  }
820 
821  if (routes.size() > 0) {
822  Route4AddEvent aev = Route4AddEvent();
823  aev.routes = routes;
824  aev.attribs = update->path_attribute;
825  config.rev_bus->publish(this, aev);
826  }
827  }
828  }
829 
830  if (send_ipv6_routes) {
831  if (update->hasAttrib(MP_UNREACH_NLRI)) {
832  const BgpPathAttrib &attr = update->getAttrib(MP_UNREACH_NLRI);
833  const BgpPathAttribMpNlriBase &mp_unreach = dynamic_cast<const BgpPathAttribMpNlriBase &>(attr);
834  if (mp_unreach.afi == IPV6 && mp_unreach.safi == UNICAST) {
835  const BgpPathAttribMpUnreachNlriIpv6 &unreach = dynamic_cast<const BgpPathAttribMpUnreachNlriIpv6 &>(mp_unreach);
836  rib6->withdraw(peer_bgp_id, unreach.withdrawn_routes);
837 
838  if (rev_bus_exist) {
839  Route6WithdrawEvent wev = Route6WithdrawEvent();
840  wev.routes = unreach.withdrawn_routes;
841  config.rev_bus->publish(this, wev);
842  }
843  }
844  }
845 
846  if (update->hasAttrib(MP_REACH_NLRI)) {
847  const BgpPathAttrib &attr = update->getAttrib(MP_REACH_NLRI);
848  const BgpPathAttribMpNlriBase &mp_reach = dynamic_cast<const BgpPathAttribMpNlriBase &>(attr);
849  if (mp_reach.afi == IPV6 && mp_reach.safi == UNICAST) {
850  const BgpPathAttribMpReachNlriIpv6 &reach = dynamic_cast<const BgpPathAttribMpReachNlriIpv6 &>(mp_reach);
851 
852  // filter toures
853  std::vector<Prefix6> filtered_routes;
854  for (const Prefix6 &route : reach.nlri) {
855  if (config.in_filters6.apply(route) == ACCEPT) filtered_routes.push_back(route);
856  }
857 
858  if (filtered_routes.size() <= 0) return 1;
859 
860  // remove MP_* & nexthop attribute
861  std::vector<std::shared_ptr<BgpPathAttrib>> attrs;
862 
863  for (const std::shared_ptr<BgpPathAttrib> &attr : update->path_attribute) {
864  if (attr->type_code == MP_REACH_NLRI || attr->type_code == MP_UNREACH_NLRI || attr->type_code == NEXT_HOP) continue;
865  attrs.push_back(attr);
866  }
867 
868  // TODO: validAddr6 to check nexthop
869 
870  rib6->insert(peer_bgp_id, filtered_routes, reach.nexthop_global, reach.nexthop_linklocal, attrs);
871 
872  if (rev_bus_exist) {
873  Route6AddEvent aev = Route6AddEvent();
874  aev.attribs = attrs;
875  aev.routes = filtered_routes;
876  config.rev_bus->publish(this, aev);
877  }
878  }
879  }
880  }
881 
882  return 1;
