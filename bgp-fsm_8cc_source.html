152 int BgpFsm::run(const uint8_t *buffer, const size_t buffer_size) {
153  if (state == BROKEN) {
154  logger->log(ERROR, "BgpFsm::run: FSM is broken, consider reset.\n");
155  return -1;
156  }
157 
158  ssize_t fill_ret = in_sink.fill(buffer, buffer_size);
159  if (fill_ret != (ssize_t) buffer_size) {
160  logger->log(ERROR, "BgpFsm::run: failed to fill() sink.\n");
161  setState(BROKEN);
162  return -1;
163  }
164 
165  // tick the clock
166  int tick_ret = tick();
167  if (tick_ret <= 0) return tick_ret;
168  last_recv = clock->getTime();
169 
170  int final_ret_val = -1;
171 
172  // keep running untill sink empty
173  while (in_sink.getBytesInSink() > 0) {
174  BgpPacket *packet = NULL;
175  ssize_t poured = in_sink.pour(&packet);
176 
177  if (poured <= -2) {
178  logger->log(ERROR, "BgpFsm::run: sink seems to be broken, please reset.\n");
179  setState(BROKEN);
180  return -1;
181  }
182 
183  if (poured == 0) return 3;
184 
185  LIBBGP_LOG(logger, DEBUG) {
186  logger->log(DEBUG, "BgpFsm::run: got message (Current state: %s):\n", bgp_fsm_state_str[state]);
187  logger->log(DEBUG, *packet);
188  }
189 
190  const BgpMessage *msg = packet->getMessage();
191  // parse failed / packet invalid (errors like Unsupported Optional
192  // Parameter falls in this catagory, since those errors are checked by
193  // parsers, other errors like FSM error, Bad Peer AS, etc is handled in
194  // fsmEval*)
195  if (poured == -1) {
196  if (msg->type == NOTIFICATION) {
197  logger->log(ERROR, "BgpFsm::run: got invalid NOTIFICATION message.\n");
198  if (state == ESTABLISHED) {
199  logger->log(ERROR, "BgpFsm::run: discarding all routes.\n");
200  }
201 
202  delete packet;
203  setState(IDLE);
204  return 0;
205  }
206  BgpNotificationMessage notify (logger, msg->getErrorCode(), msg->getErrorSubCode(), msg->getError(), msg->getErrorLength());
207  setState(IDLE);
208  if(!writeMessage(notify)) return -1;
209  delete packet;
210  return 0;
211  }
212 
213  if (msg->type == NOTIFICATION) {
214  const BgpNotificationMessage *notify = dynamic_cast<const BgpNotificationMessage *>(msg);
215  const char *err_msg = bgp_error_code_str[notify->errcode];
216  const char *err_sub_msg = bgp_error_code_str[0];
217  switch (notify->errcode) {
218  case E_HEADER: err_sub_msg = bgp_header_error_subcode_str[notify->subcode]; break;
219  case E_OPEN: err_sub_msg = bgp_open_error_subcode_str[notify->subcode]; break;
220  case E_UPDATE: err_sub_msg = bgp_update_error_str[notify->subcode]; break;
221  case E_FSM: err_sub_msg = bgp_fsm_error_str[notify->subcode]; break;
222  case E_CEASE: err_sub_msg = bgp_cease_error_str[notify->subcode]; break;
223  }
224  logger->log(ERROR, "BgpFsm::run: got NOTIFICATION: %s (%d): %s (%d).\n", err_msg, notify->errcode, err_sub_msg, notify->subcode);
225  delete packet;
226  setState(IDLE);
227  return 0;
228  }
229 
230  int retval = -1;
231 
232  int vald_ret = validateState(msg->type);
233  if (vald_ret <= 0) {
234  delete packet;
235  return vald_ret;
236  }
237 
238  switch (state) {
239  case IDLE: retval = fsmEvalIdle(msg); break;
240  case OPEN_SENT: retval = fsmEvalOpenSent(msg); break;
241  case OPEN_CONFIRM: retval = fsmEvalOpenConfirm(msg); break;
242  case ESTABLISHED: retval = fsmEvalEstablished(msg); break;
243  default: {
244  logger->log(ERROR, "BgpFsm::run: FSM in invalid state: %d.\n", state);
245  delete packet;
246  return -1;
247  }
248  }
249 
250  delete packet;
251  if (retval < 0) return retval;
252  if (retval == 0) final_ret_val = 0;
253  if (retval == 1 && final_ret_val != 0 && final_ret_val != 2) final_ret_val = 1;
254  if (retval == 2) final_ret_val = 2;
255  }
256 
257  return final_ret_val;
258 }

294 int BgpFsm::openRecv(const BgpOpenMessage *open_msg) {
295  if (open_msg->version != 4) {
296  BgpNotificationMessage notify (logger, E_OPEN, E_VERSION, NULL, 0);
297  setState(IDLE);
298  if(!writeMessage(notify)) return -1;
299  return 0;
300  }
301 
302  if (config.peer_asn != 0 && open_msg->my_asn != config.peer_asn) {
303  BgpNotificationMessage notify (logger, E_OPEN, E_PEER_AS, NULL, 0);
304  setState(IDLE);
305  if(!writeMessage(notify)) return -1;
306  return 0;
307  }
308 
309  if (!validAddr(open_msg->bgp_id)) {
310  LIBBGP_LOG(logger, ERROR) {
311  char ip_str[INET_ADDRSTRLEN];
312  inet_ntop(AF_INET, &(open_msg->bgp_id), ip_str, INET_ADDRSTRLEN);
313  logger->log(ERROR, "BgpFsm::openRecv: peer BGP ID (%s) invalid.\n", ip_str);
314  }
315  BgpNotificationMessage notify (logger, E_OPEN, E_BGP_ID, NULL, 0);
316  setState(IDLE);
317  if(!writeMessage(notify)) return -1;
318  return 0;
319  }
320 
321  // if hold timer != 0 but < 3, reject witl E_HOLD_TIME (as per rfc4271)
322  if (open_msg->hold_time < 3 && open_msg->hold_time != 0) {
323  logger->log(ERROR, "BgpFsm::openRecv: invalid hold timer %d.\n", open_msg->hold_time);
324  BgpNotificationMessage notify (logger, E_OPEN, E_HOLD_TIME, NULL, 0);
325  setState(IDLE);
326  if(!writeMessage(notify)) return -1;
327  return 0;
328  }
329 
330  if (!config.no_collision_detection && rev_bus_exist) {
331  RouteCollisionEvent col = RouteCollisionEvent();
332  col.peer_bgp_id = open_msg->bgp_id;
333 
334  // publish returned 0: no one complained about collision (either
335  // there's no collision, or some fsm has killed themselves)
336  //
337  // publish returned > 0: someone complained about collision, it think
338  // this session should be dropped
339  if(config.rev_bus->publish(this, col) > 0) {
340  int res_result = resloveCollision(open_msg->bgp_id, true);
341 
342  if(res_result == -1) return -1;
343  if(res_result == 0) return 0;
344 
345  // should not be happening
346  if(res_result == 1) {
347  logger->log(ERROR,
348  "BgpFsm::openRecv: collision found, and some other FSM feels like they should live"
349  "while we feel like we should live too. is there duplicated FSMs?\n"
350  );
351  setState(BROKEN);
352  return -1;
353  }
354  }
355  }
356 
357  hold_timer = config.hold_timer > open_msg->hold_time ? open_msg->hold_time : config.hold_timer;
358  peer_bgp_id = open_msg->bgp_id;
359  peer_asn = open_msg->getAsn();
360  use_4b_asn = open_msg->hasCapability(ASN_4B) && config.use_4b_asn;
361 
362  return 1;
363 }

365 int BgpFsm::resloveCollision(uint32_t peer_bgp_id, bool is_new) {
366  if(is_new) {
367  if (ntohl(config.router_id) > ntohl(peer_bgp_id)) {
368  // this is a new connection, and "we" have higer ID, there's
369  // already a connection so THIS fsm should be dispose. since
370  // THIS fsm is created by peer connecting to us.
371  BgpNotificationMessage notify (logger, E_CEASE, E_COLLISION, NULL, 0);
372  if(!writeMessage(notify)) return -1;
373 
374  setState(IDLE);
375  return 0;
376  } else {
377  // this is a new connection, and peer has higher ID. the exisiting
378  // connection should be dispose, since THIS fsm is created by peer
379  // connecting to us, this one will be kept, we do nothing.
380  return 1;
381  }
382  } else {
383  if (config.router_id > peer_bgp_id) {
384  // this is a old connection, and "we" have higer ID, the new one
385  // shoud close, we do nothing.
386 
387  return 1;
388  } else {
389  // this is a old connection, and "peer" have higer ID, this one
390  // shoud close.
391  BgpNotificationMessage notify (logger, E_CEASE, E_COLLISION, NULL, 0);
392  if(!writeMessage(notify)) return -1;
393 
394  setState(IDLE);
395  return 0;
396  }
397  }
398 
399  // UNREACHED
400  setState(BROKEN);
401  logger->log(ERROR, "BgpFsm::resloveCollison: ??? :( \n");
402  return -1;
403 }

465  update.dropNonTransitive();
466  if (config.forced_default_nexthop || !update.hasAttrib(NEXT_HOP)) {
467  LIBBGP_LOG(logger, INFO) {
468  char ip_str[INET_ADDRSTRLEN];
469  inet_ntop(AF_INET, &(config.nexthop), ip_str, INET_ADDRSTRLEN);
470  if (config.forced_default_nexthop) {
471  logger->log(INFO, "BgpFsm::prepareUpdateMessage: forced_default_nexthop set, using %s as nexthop.\n", ip_str);
472  } else {
473  logger->log(INFO, "BgpFsm::prepareUpdateMessage: no nethop attribute set, using %s as nexthop.\n", ip_str);
474  }
475  }
476  update.setNextHop(config.nexthop);
477  } else {
478  BgpPathAttribNexthop &nh = dynamic_cast<BgpPathAttribNexthop &> (update.getAttrib(NEXT_HOP));
479  if (!Route::Includes(config.peering_lan_prefix, config.peering_lan_length, nh.next_hop)) {
480  LIBBGP_LOG(logger, INFO) {
481  char def_nexthop[INET_ADDRSTRLEN];
482  char cur_nexthop[INET_ADDRSTRLEN];
483  inet_ntop(AF_INET, &(config.nexthop), def_nexthop, INET_ADDRSTRLEN);
484  inet_ntop(AF_INET, &(nh.next_hop), cur_nexthop, INET_ADDRSTRLEN);
485  logger->log(INFO, "BgpFsm::prepareUpdateMessage: nexthop %s not in peering lan, using %s.\n", cur_nexthop, def_nexthop);
486  }
487  nh.next_hop = config.nexthop;
488  }
489  }
490 
491  if (config.use_4b_asn && use_4b_asn) {
492  update.restoreAsPath();
493  update.restoreAggregator();
494  update.prepend(config.asn);
495  } else {
496  update.downgradeAsPath();
497  update.downgradeAggregator();
498  update.prepend(config.asn);
499  }

507  return 0;
508  }
509  return 1;
510  case OPEN_SENT:
511  if (type != OPEN) {
512  logger->log(ERROR, "BgpFsm::validateState: got non-OPEN message in OPEN_SENT state.\n");

514  setState(IDLE);

516 
517  return 0;
518  }
519  return 1;
520  case OPEN_CONFIRM:
521  if (type != KEEPALIVE) {

523  BgpNotificationMessage notify (logger, E_FSM, E_OPEN_CONFIRM, NULL, 0);
524  setState(IDLE);
525  if(!writeMessage(notify)) return -1;
526 
527  return 0;
528  }
529  return 1;
530  case ESTABLISHED:
531  if (type != UPDATE && type != KEEPALIVE) {
532  logger->log(ERROR, "BgpFsm::validateState: got invalid message (type %d) in ESTABLISHED state.\n", type);
533  BgpNotificationMessage notify (logger, E_FSM, E_ESTABLISHED, NULL, 0);
534  setState(IDLE);
535  if(!writeMessage(notify)) return -1;
536 
537  return 0;
538  }
539  return 1;
540  default:
541  logger->log(ERROR, "BgpFsm::validateState: got message in bad state. consider reset.\n");

627  for (const Route &route : update->withdrawn_routes) {
628  rib->withdraw(peer_bgp_id, route);
629  }
630 
631  if (update->nlri.size() > 0) {
632  const BgpPathAttribNexthop &nh = dynamic_cast<const BgpPathAttribNexthop &>(update->getAttrib(NEXT_HOP));
633 
634  if (!validAddr(nh.next_hop)) {
635  LIBBGP_LOG(logger, WARN) {
636  char ip_str[INET_ADDRSTRLEN];
637  inet_ntop(AF_INET, &(nh.next_hop), ip_str, INET_ADDRSTRLEN);
638  logger->log(WARN, "BgpFsm::fsmEvalEstablished: ignored %zu routes with invalid nexthop %s\n", update->nlri.size(), ip_str);
639  }
640  return 1;
641  }
642 
643  if (!config.no_nexthop_check && !Route::Includes(config.peering_lan_prefix, config.peering_lan_length, nh.next_hop)) {
644  LIBBGP_LOG(logger, WARN) {
645  char ip_str_nh[INET_ADDRSTRLEN];
646  char ip_str_lan[INET_ADDRSTRLEN];
647  inet_ntop(AF_INET, &(nh.next_hop), ip_str_nh, INET_ADDRSTRLEN);
648  inet_ntop(AF_INET, &(config.peering_lan_prefix), ip_str_lan, INET_ADDRSTRLEN);
649  logger->log(WARN, "BgpFsm::fsmEvalEstablished: ignored %zu routes with nexthop outside peering LAN. (%s not in %s/%d)\n",
650  update->nlri.size(), ip_str_nh, ip_str_lan, config.peering_lan_length);
651  }
652  return 1;
653  };
654  }
655 
656 
657  std::vector<Route> routes = std::vector<Route> ();
658  for (const Route &route : update->nlri) {
659  if(config.in_filters.apply(route.getPrefix(), route.getLength()) == ACCEPT) {
660  routes.push_back(route);
661  } else {
662  LIBBGP_LOG(logger, INFO) {
663  uint32_t prefix = route.getPrefix();
664  char ip_str[INET_ADDRSTRLEN];
665  inet_ntop(AF_INET, &prefix, ip_str, INET_ADDRSTRLEN);
666  logger->log(INFO, "BgpFsm::fsmEvalEstablished: route %s/%d filtered by in_filters.\n", ip_str, route.getLength());
667  }
668  }
669  }
670 
671  if (routes.size() > 0) {
672  rib->insert(peer_bgp_id, routes, update->path_attribute);
673  }
674 
675  if (rev_bus_exist) {
676  if (update->withdrawn_routes.size() > 0) {
677  RouteWithdrawEvent wev = RouteWithdrawEvent();
678  wev.routes = update->withdrawn_routes;
679  config.rev_bus->publish(this, wev);
680  }
681 
682  if (routes.size() > 0) {
683  RouteAddEvent aev = RouteAddEvent();
684  aev.routes = routes;
685  aev.attribs = update->path_attribute;
686  config.rev_bus->publish(this, aev);
687  }
688  }
689 
690  return 1;
691 }

719 bool BgpFsm::validAddr(uint32_t addr) const {
720  if (addr == config.nexthop || addr == config.router_id) {
721  return false;
722  }
723 
724  uint32_t addr_host = ntohl(addr);
725  uint32_t first = addr_host >> 24;
726 
727  if (first == 0 || first == 127 || (first >= 224 && first <= 239) || first > 240) {
728  return false;
729  }
730 
731  return true;
732 }
