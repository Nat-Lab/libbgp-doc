113 const BgpRib4Entry* BgpRib4::insert(BgpLogHandler *logger, const Prefix4 &route, uint32_t nexthop, uint32_t weight) {
114  std::vector<std::shared_ptr<BgpPathAttrib>> attribs;
115  BgpPathAttribOrigin *origin = new BgpPathAttribOrigin(logger);
116  BgpPathAttribNexthop *nexhop_attr = new BgpPathAttribNexthop(logger);
117  BgpPathAttribAsPath *as_path = new BgpPathAttribAsPath(logger, true);
118  nexhop_attr->next_hop = nexthop;
119  origin->origin = IGP;
120 
121  attribs.push_back(std::shared_ptr<BgpPathAttrib>(origin));
122  attribs.push_back(std::shared_ptr<BgpPathAttrib>(nexhop_attr));
123  attribs.push_back(std::shared_ptr<BgpPathAttrib>(as_path));
124 
125  uint64_t use_update_id = update_id;
126 
127  for (const BgpRib4Entry &entry : rib) {
128  if (entry.src_router_id == 0 && entry.route == route) {
129  logger->log(ERROR, "BgpRib4::insert: route exists.");
130  return NULL;
131  }
132 
133  // see if we can group this entry to other local entries
134  if (entry.src_router_id == 0) {
135  for (const std::shared_ptr<BgpPathAttrib> &attr : entry.attribs) {
136  if (attr->type_code == NEXT_HOP) {
137  const BgpPathAttribNexthop &nh = dynamic_cast<const BgpPathAttribNexthop &>(*attr);
138  if (nh.next_hop == nexthop) use_update_id = entry.update_id;
139  }
140  }
141  }
142  }
143 
144  BgpRib4Entry new_entry(route, 0, attribs);
145  std::lock_guard<std::recursive_mutex> lock(mutex);
146  new_entry.update_id = use_update_id;
147  new_entry.weight = weight;
148  if (use_update_id == update_id) update_id++;
149  rib.push_back(new_entry);
150 
151  return &(rib.back());
152 }
