270 const BgpRib4Entry* BgpRib4::selectEntry(const BgpRib4Entry *a, const BgpRib4Entry *b) {
271  if (a == NULL) return b;
272  if (b == NULL) return a;
273 
274  const Prefix4 &ra = a->route;
275  const Prefix4 &rb = b->route;
276 
277  // a is more specific, use a
