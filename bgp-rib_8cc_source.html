214 const BgpRibEntry* BgpRib::selectEntry(const BgpRibEntry *a, const BgpRibEntry *b) {
215  assert(!(a == NULL && b == NULL));
216 
217  if (a == NULL) return b;
218  if (b == NULL) return a;
219 
220  const Route &ra = a->route;
221  const Route &rb = b->route;
222 
223  // a is more specific, use a

225 
226  // a, b are same level of specific, check metric

230  }

232  // b is more specific, use b
