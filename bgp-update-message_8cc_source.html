178 bool BgpUpdateMessage::prepend(uint32_t asn) {

181 
182  // AS4_PATH can't exist in 4b mode

185  "consider restoreAsPath().\n");

187  }
188

190  BgpPathAttribAsPath *path = new BgpPathAttribAsPath(use_4b_asn);
191  path->prepend(asn);

193  return true;
194  }
195

197  if (!path.is_4b) {

205  // in 2b-mode, prepend 2b asn to AS_PATH and update AS4_PATH.

207  // FIXME: don't change as4_path if both side disabled 4b support
208 
209  uint16_t prep_asn = asn >= 0xffff ? 23456 : asn;
210 
211  if (!hasAttrib(AS_PATH)) {
212  BgpPathAttribAsPath *path = new BgpPathAttribAsPath(use_4b_asn);
213  path->prepend(prep_asn);

215  } else {

219  "consider downgradeAsPath().\n");

243 bool BgpUpdateMessage::restoreAsPath() {
244  if (!hasAttrib(AS_PATH)) return true;
245 
246  BgpPathAttribAsPath &path = dynamic_cast<BgpPathAttribAsPath &>(getAttrib(AS_PATH));
247  if (path.is_4b) return true;
248 
249  // no AS4_PATH, just make AS_PATH 4b
250  if (!hasAttrib(AS4_PATH)) {
251  std::vector<BgpAsPathSegment> new_segs;
252 
253  for (const BgpAsPathSegment &seg2 : path.as_paths) {
254  if (seg2.is_4b) {
255  _bgp_error("BgpUpdateMessage::restoreAsPath: 4b seg found in 2b attrib.\n");
256  return false;
257  }
258 
259  BgpAsPathSegment new_seg (true, seg2.type);
260  for (uint16_t asn : seg2.value) {
261  if (asn == 23456) {
262  _bgp_error("BgpUpdateMessage::restoreAsPath: warning: AS_TRANS found but no AS4_PATH.\n");
263  }
264  new_seg.value.push_back(asn);
265  }
266 
267  new_segs.push_back(new_seg);
268  }
269 
270  path.as_paths = new_segs;
271  path.is_4b = true;
272  return true;
273  }
274 
275  // we have AS4_PATH recorver AS_TRANS.
276  std::vector<uint32_t> full_as_path;
277  const BgpPathAttribAs4Path &as4_path = dynamic_cast<const BgpPathAttribAs4Path &>(getAttrib(AS4_PATH));
278  for (const BgpAsPathSegment &seg4 : as4_path.as4_paths) {
279  if (!seg4.is_4b) {
280  _bgp_error("BgpUpdateMessage::restoreAsPath: bad as4_path: found 2b seg.\n");
281  return false;
282  }
283 
284  if (seg4.type == AS_SEQUENCE) {
285  const std::vector<uint32_t> &part = seg4.value;
286  full_as_path.insert(full_as_path.end(), part.begin(), part.end());
287  }
288  }
289 
290  // AS4_PATH should be removed.
291  dropAttrib(AS4_PATH);
292 
293  bool has_4b = false;
294 
295  // find the iterator to first asn > 0xffff
296  std::vector<uint32_t>::const_iterator iter_4b = full_as_path.begin();
297 
298  for(; iter_4b != full_as_path.end(); iter_4b++) {
299  has_4b = true;
300  if (*iter_4b > 0xffff) break;
301  }
302 
303  std::vector<BgpAsPathSegment> new_segs;
304 
305  for (const BgpAsPathSegment &seg2 : path.as_paths) {
306  std::vector<uint32_t>::const_iterator local_iter = iter_4b;
307  if (seg2.is_4b) {
308  _bgp_error("BgpUpdateMessage::restoreAsPath: 4b seg found in 2b attrib.\n");
309  return false;
310  }
311 
312  BgpAsPathSegment new_seg (true, seg2.type);
313 
314  // increment the local_iter iterator?
315  bool incr_iter = false;
316  for (uint16_t asn : seg2.value) {
317  uint32_t new_asn = asn;
318 
319  // AS4_PATH avaliale & not ended?
320  if (has_4b && local_iter != full_as_path.end()) {
321 
322  // we found AS_TRANS, we need to replace it with last asn in AS_TRANS
323  if (new_asn == 23456) {
324 
325  // we have hit our first AS_TRANS. from now on, we need to move the AS4_PATH
326  // iterator too.
327  incr_iter = true;
328  new_asn = *local_iter;
329  } else if (new_asn != *local_iter) {
330  _bgp_error("BgpUpdateMessage::restoreAsPath: warning: AS_PATH and AS4_PATH does not match.\n");
331  }
332 
333  if (incr_iter) local_iter++;
334  }
335 
336  new_seg.value.push_back(new_asn);
337  }
338 
339  new_segs.push_back(new_seg);
340  }
341 
342  path.is_4b = true;
343  path.as_paths = new_segs;
344  return true;
345

507 bool BgpUpdateMessage::validateAttribs() {
508  bool has_origin = false;
509  bool has_nexthop = false;
510  bool has_as_path = false;
511 
512  uint32_t typecode_bitsmap = 0;
513 
514  for (std::vector<std::shared_ptr<BgpPathAttrib>>::const_iterator attr_iter = path_attribute.begin();
515  attr_iter != path_attribute.end(); attr_iter++) {
516  uint8_t type_code = (*attr_iter)->type_code;
517 
518  if (type_code == AS_PATH) has_as_path = true;
519  else if (type_code == NEXT_HOP) has_nexthop = true;
520  else if (type_code == ORIGIN) has_origin = true;
521 
522  if ((typecode_bitsmap >> type_code) & 1U) {
523  _bgp_error("BgpUpdateMessage::validateAttribs:: duplicated attribute type in list: %d\n", type_code);
524  setError(E_UPDATE, E_ATTR_LIST, NULL, 0);
525  return false;
526  }
527 
528  typecode_bitsmap &= ~(1UL << type_code);
529  }
530 
531  if (!(has_as_path && has_nexthop && has_origin)) {
532  _bgp_error("BgpUpdateMessage::validateAttribs: mandatory attribute(s) missing.\n");
533  setError(E_UPDATE, E_MISS_WELL_KNOWN, NULL, 0);
534  return false;
535  }
536 
537  return true;
538 }

540 ssize_t BgpUpdateMessage::parse(const uint8_t *from, size_t msg_sz) {
541  if (msg_sz < 4) {
542  uint8_t _err_data = msg_sz;
543  setError(E_HEADER, E_LENGTH, &_err_data, sizeof(uint8_t));
544  _bgp_error("BgpUpdateMessage::parse: invalid open message size: %d.\n", msg_sz);
545  return -1;
546  }
547 
548  const uint8_t *buffer = from;
549 
550  uint16_t withdrawn_len = ntohs(getValue<uint16_t>(&buffer)); // len: 2
551 
552  if (withdrawn_len > msg_sz - 4) { // -4: two length fields (withdrawn len + attrib len)
553  _bgp_error("BgpUpdateMessage::parse: withdrawn routes length overflows message.\n");
554  setError(E_UPDATE, E_UNSPEC, NULL, 0);
555  return -1;
556  }
557 
558  uint16_t parsed_withdrawn_len = 0;
559 
560  while (parsed_withdrawn_len < withdrawn_len) {
561  if (withdrawn_len - parsed_withdrawn_len < 1) {
562  _bgp_error("BgpUpdateMessage::parse: unexpected end of withdrawn routes list.\n");
563  setError(E_UPDATE, E_UNSPEC, NULL, 0);
564  return -1;
565  }
566 
567  uint8_t route_len = getValue<uint8_t>(&buffer); // len2: 1
568  parsed_withdrawn_len++;
569  if (route_len > 32) {
570  _bgp_error("BgpUpdateMessage::parse: invalid route len in withdrawn routes: %d\n", route_len);
571  setError(E_UPDATE, E_UNSPEC, NULL, 0);
572  return -1;
573  }
574 
575  size_t route_buffer_len = (route_len + 7) / 8;
576  if (parsed_withdrawn_len + route_buffer_len > withdrawn_len) {
577  _bgp_error("BgpUpdateMessage::parse: withdrawn route overflows routes list.\n");
578  setError(E_UPDATE, E_UNSPEC, NULL, 0);
579  return -1;
580  }
581 
582  uint32_t prefix = 0;
583  memcpy(&prefix, buffer, route_buffer_len);
584  Route route(prefix, route_len);
585  withdrawn_routes.push_back(route);
586 
587  buffer += route_buffer_len; // len2: route_buffer_len
588  parsed_withdrawn_len += route_buffer_len;
589  }
590 
591  assert(parsed_withdrawn_len == withdrawn_len);
592 
593  uint16_t attribute_len = ntohs(getValue<uint16_t>(&buffer)); // len: 2
594  if ((size_t) (attribute_len + withdrawn_len + 4) > msg_sz) {
595  _bgp_error("BgpUpdateMessage::parse: attribute list length overflows message buffer.\n");
596  setError(E_UPDATE, E_ATTR_LIST, NULL, 0);
597  return -1;
598  }
599 
600  uint16_t parsed_attribute_len = 0;
601 
602  while (parsed_attribute_len < attribute_len) {
603  if (attribute_len - parsed_attribute_len < 3) {
604  _bgp_error("BgpUpdateMessage::parse: unexpected end of attribute list.\n");
605  setError(E_UPDATE, E_UNSPEC, NULL, 0);
606  return -1;
607  }
608 
609  int8_t attr_type = BgpPathAttrib::GetTypeFromBuffer(buffer, attribute_len - parsed_attribute_len);
610 
611  if (attr_type < 0) {
612  _bgp_error("BgpUpdateMessage::parse: failed to parse attribute type.\n");
613  setError(E_UPDATE, E_UNSPEC, NULL, 0);
614  return -1;
615  }
616 
617  BgpPathAttrib *attrib = NULL;
618 
619  switch(attr_type) {
620  case ORIGIN: attrib = new BgpPathAttribOrigin(); break;
621  case AS_PATH: attrib = new BgpPathAttribAsPath(use_4b_asn); break;
622  case NEXT_HOP: attrib = new BgpPathAttribNexthop(); break;
623  case MULTI_EXIT_DISC: attrib = new BgpPathAttribMed(); break;
624  case LOCAL_PREF: attrib = new BgpPathAttribLocalPref(); break;
625  case ATOMIC_AGGREGATE: attrib = new BgpPathAttribAtomicAggregate(); break;
626  case AGGREATOR: attrib = new BgpPathAttribAggregator(use_4b_asn); break;
627  case COMMUNITY: attrib = new BgpPathAttribCommunity(); break;
628  case AS4_PATH: attrib = new BgpPathAttribAs4Path(); break;
629  case AS4_AGGREGATOR: attrib = new BgpPathAttribAs4Aggregator(); break;
630  default: attrib = new BgpPathAttrib(); break;
631  }
632 
633  assert(attrib != NULL);
634 
635  ssize_t attrib_parsed = attrib->parse(buffer, attribute_len - parsed_attribute_len);
636 
637  if (attrib_parsed < 0) {
638  forwardParseError(*attrib);
639  delete attrib;
640  return -1;
641  }
642 
643  buffer += attrib_parsed;
644  parsed_attribute_len += attrib_parsed;
645  path_attribute.push_back(std::shared_ptr<BgpPathAttrib>(attrib));
646  }
647 
648  if (!validateAttribs()) return -1;
649 
650  assert(parsed_attribute_len == attribute_len);
651 
652  // 4: len fields (withdrawn len & attrib len)
653  size_t nlri_len = msg_sz - 4 - parsed_attribute_len - parsed_withdrawn_len;
654  size_t parsed_nlri_len = 0;
655 
656  while (parsed_nlri_len < nlri_len) {
657  if (nlri_len - parsed_nlri_len < 1) {
658  _bgp_error("BgpOpenMessage::parse: unexpected end of nlri.\n");
659  setError(E_UPDATE, E_UNSPEC, NULL, 0);
660  return -1;
661  }
662 
663  uint8_t route_len = getValue<uint8_t>(&buffer); // len2: 1
664  parsed_nlri_len++;
665  if (route_len > 32) {
666  _bgp_error("BgpUpdateMessage::parse: invalid route len in nlri routes: %d\n", route_len);
667  setError(E_UPDATE, E_UNSPEC, NULL, 0);
668  return -1;
669  }
670 
671  size_t route_buffer_len = (route_len + 7) / 8;
672  if (parsed_nlri_len + route_buffer_len > nlri_len) {
673  _bgp_error("BgpUpdateMessage::parse: nlri route overflows routes list.\n");
674  setError(E_UPDATE, E_UNSPEC, NULL, 0);
675  return -1;
676  }
677 
678  uint32_t prefix = 0;
679  memcpy(&prefix, buffer, route_buffer_len);
680  Route route(prefix, route_len);
681  nlri.push_back(route);
682 
683  buffer += route_buffer_len; // len2: route_buffer_len
684  parsed_nlri_len += route_buffer_len;
685  }
686 
687  assert(parsed_nlri_len + parsed_attribute_len + parsed_withdrawn_len + 4 == msg_sz);
688 
689  return msg_sz;
690 }

692 ssize_t BgpUpdateMessage::write(uint8_t *to, size_t buf_sz) const {
693  if (buf_sz < 4) {
694  _bgp_error("BgpUpdateMessage::write: destination buffer too small.\n");
695  return -1;
696  }
697 
698  size_t tot_written = 0;
699  uint8_t *buffer = to;
700 
701  // keep a pointer to len field to write length to later
702  uint8_t *withdrawn_routes_len_ptr = buffer;
703  buffer++; // skip the length field for now
704 
705  size_t written_withdrawn_length = 0;
706 
707  for (const Route &route : withdrawn_routes) {
708  uint8_t route_len = route.getLength();
709  size_t pfx_buf_sz = (route_len + 7) / 8;
710 
711  // 1: this prefix len field
712  if (1 + written_withdrawn_length + pfx_buf_sz + tot_written > buf_sz) {
713  _bgp_error("BgpUpdateMessage::write: destination buffer too small.\n");
714  return -1;
715  }
716 
717  putValue<uint8_t>(&buffer, route_len);
718  memcpy(buffer, &route_len, pfx_buf_sz);
719 
720  buffer += pfx_buf_sz;
721  written_withdrawn_length += 1 + pfx_buf_sz;
722  }
723 
724  // now, put the length
725  putValue<uint16_t>(&withdrawn_routes_len_ptr, htons(written_withdrawn_length));
726 
727  tot_written += written_withdrawn_length + 2; // + 2: the length field
728 
729  // keep a pointer to len field to write length to later
730  uint8_t *route_attrib_len_ptr = buffer;
731  buffer++;

736  ssize_t buf_left = buf_sz - written_attrib_length - tot_written;
737  assert(buf_left >= 0);
738

740 
741  if (write_ret < 0) return -1;
742  written_attrib_length += write_ret;
743  }
744 
745  // put the length

759  _bgp_error("BgpUpdateMessage::write: destination buffer too small.\n");
760  return -1;
